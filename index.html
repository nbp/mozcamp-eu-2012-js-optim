  • Not with indexed properties.
  • Forbid JIT compilation.

with (o) { b }; // = 0
with (o) { 0 }; // = 0

0; /* = 0 */   o[0];   /* = "zero" */

Int Operations

Additions: b + c | 0

Multiplication: not b * c | 0, but 


function mul32(a, b) {
  var a32 = a | 0, b32 = b | 0,
      alo = a & 0xffff,
      ahi = a32 - alo | 0,
      rlo = alo * b32 | 0;
      rhi = ahi * b32 | 0;
  return rhi + rlo | 0;
}
